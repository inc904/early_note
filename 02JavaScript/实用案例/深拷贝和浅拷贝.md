- 浅拷贝只是拷贝一层，更深层次对象级别的只是拷贝引用

- 深拷贝拷贝多层，每一级别的数据都会拷贝。

- Object.assign(newObj, oldObj) 直接实现浅拷贝

- 深拷贝

  ```js
  // 实现深拷贝
  var obj = {
      id: 1,
      name: 'andy',
      msg: {
          age: 18
      },
      color: ['red', 'pink']
  }
  var o = {}
  function deepCopy(newObj, oldObj) {
      for (var k in oldObj) {
          // 判断数据类型 k 是属性名 obj[k] 是属性值
          // 获取属性值
          var item = oldObj[k]
          // 判断数据类型
          if (item instanceof Array) {
              newObj[k] = []
              deepCopy(newObj[k], item)
          } else if (item instanceof Object) {
              newObj[k] = {}
              deepCopy(newObj[k], item)
          } else {
              newObj[k] = item
          }
      }
  }
  deepCopy(o, obj)
  ```
